    <script>
        // Set current day and date
        const days = ['Minggu', 'Senin', 'Selasa', 'Rabu', 'Kamis', 'Jumat', 'Sabtu'];
        const now = new Date();
        document.getElementById('hari').textContent = `Hari: ${days[now.getDay()]}`;
        document.getElementById('tanggal').textContent = `Tanggal: ${now.getDate()}/${now.getMonth() + 1}/${now.getFullYear()}`;

        // Function to format number with thousand separator and Rupiah
        function formatRupiah(num) {
            return num.toLocaleString('id-ID', { style: 'currency', currency: 'IDR' }).replace(/,00$/, '');
        }

        // Function to parse input value
        function parseInput(value) {
            return parseFloat(value.replace(/[^\d]/g, '')) || 0;
        }

        // Calculate results and total
        document.getElementById('salesForm').addEventListener('submit', function(e) {
            e.preventDefault();

            const bca = parseInput(document.getElementById('bca').value);
            const ovo = parseInput(document.getElementById('ovo').value);
            const ovoResult = ovo * 0.993;
            document.getElementById('ovoResult').textContent = formatRupiah(ovoResult);

            const grabfood = parseInput(document.getElementById('grabfood').value);
            const grabfoodResult = grabfood * 0.63;
            document.getElementById('grabfoodResult').textContent = formatRupiah(grabfoodResult);

            const shopeefood = parseInput(document.getElementById('shopeefood').value);
            const shopeefoodResult = shopeefood * 0.78;
            document.getElementById('shopeefoodResult').textContent = formatRupiah(shopeefoodResult);

            const gofood = parseInput(document.getElementById('gofood').value);
            const gofoodResult = gofood * 0.8 - 1000;
            document.getElementById('gofoodResult').textContent = formatRupiah(gofoodResult);

            const gobiz = parseInput(document.getElementById('gobiz').value);
            const gobizResult = gobiz * 0.993;
            document.getElementById('gobizResult').textContent = formatRupiah(gobizResult);

            const uangKas = parseInput(document.getElementById('uangKas').value);

            const totalSales = bca + ovoResult + grabfoodResult + shopeefoodResult + gofoodResult + gobizResult + uangKas;
            document.getElementById('totalSalesReal').textContent = formatRupiah(totalSales);
            document.getElementById('totalSalesTablet').textContent = formatRupiah(totalSales);
        });

        // Format input values on blur
        const inputs = document.querySelectorAll('input[type="text"]');
        inputs.forEach(input => {
            input.addEventListener('blur', function() {
                this.value = formatRupiah(parseInput(this.value));
            });
        });

        // Generate report
        document.getElementById('generateReport').addEventListener('click', function() {
            const hari = document.getElementById('hari').textContent;
            const tanggal = document.getElementById('tanggal').textContent;
            const bca = document.getElementById('bca').value;
            const ovo = document.getElementById('ovo').value;
            const ovoResult = document.getElementById('ovoResult').textContent;
            const grabfood = document.getElementById('grabfood').value;
            const grabfoodResult = document.getElementById('grabfoodResult').textContent;
            const shopeefood = document.getElementById('shopeefood').value;
            const shopeefoodResult = document.getElementById('shopeefoodResult').textContent;
            const gofood = document.getElementById('gofood').value;
            const gofoodResult = document.getElementById('gofoodResult').textContent;
            const gobiz = document.getElementById('gobiz').value;
            const gobizResult = document.getElementById('gobizResult').textContent;
            const uangKas = document.getElementById('uangKas').value;
            const pengeluaranUangKas = document.getElementById('pengeluaranUangKas').value;
            const sisaUangKas = document.getElementById('sisaUangKas').value;
            const totalSalesReal = document.getElementById('totalSalesReal').textContent;
            const totalSalesTablet = document.getElementById('totalSalesTablet').textContent;
            const terjual = document.getElementById('terjual').value;

            const report = `
Pendapatan & Pengeluaran
${hari}
${tanggal}

Sales Real

1. BCA: ${bca}

2. OVO: ${ovo} - 0,7% = ${ovoResult}

3. GRABFOOD: ${grabfood} - 37% = ${grabfoodResult}

4. SHOPEEFOOD: ${shopeefood} - 22% = ${shopeefoodResult}

5. GOFOOD: ${gofood} - 20%, -1k = ${gofoodResult}

6. Gobiz: ${gobiz} - 0.7% = ${gobizResult}

7. Uang Kas: ${uangKas}

Pengeluaran =
Uang kas: ${pengeluaranUangKas}
-

Sisa uang kas: ${sisaUangKas}

Total Sales (Real) = ${totalSalesReal}
Total Sales (Tablet) = ${totalSalesTablet}

Terjual:
${terjual}

Kuota (Jumlah saat ini):
- Kentang: 3 bgks ✅
- Pop mie (semua jenis): 14 bgks ✅
            `;

            document.getElementById('report').textContent = report;
        });
